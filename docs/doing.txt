/**
 * Generates the payload for server-side requests.
 * @param {string} action - 'expand' or 'expandToLevel'
 * @param {HTMLElement|null} target - The TR element (for expand) or Control (for expandToLevel)
 * @returns {Object}
 */
_getRequestPayload(action, target = null) {
    const payload = {
        action: action,
        groupOrder: this.options.groupOrder,
        requestedTotals: this.options.totals,
        requestedColumns: this.options.columns,
        requestDetails: {
            level: -1,
            expandingLevel: 0,
            groupingDimension: null,
            displayNames: [], // Array of raw hierarchy values
            rowId: null
        }
    };

    if (action === 'expand' && target) {
        const row = target.closest('tr');
        const level = parseInt(row.getAttribute('data-level'));

        payload.requestDetails = {
            level: level,
            expandingLevel: level + 1,
            groupingDimension: row.getAttribute('data-dimension'),
            // Retrieves the complete hierarchy of raw values as an array
            displayNames: this._getHierarchyForRow(row),
            rowId: row.id
        };
    }
    else if (action === 'expandToLevel') {
        // For dimension controls, we target a specific depth in the existing groupOrder
        const level = target ? parseInt(target.dataset.level || target.value) : 0;

        payload.requestDetails = {
            level: level,
            expandingLevel: level,
            groupingDimension: this.options.groupOrder[level],
            displayNames: [], // Root level expansion starts empty
            rowId: null
        };
    }

    return payload;
}
--
data-level: (Existing) Indicates the depth (0, 1, 2...). Essential for the server to know which index of groupOrder to process next.

data-display-name: (New) Stores the raw value of the grouping key (e.g., "Electronics"). This is critical because the innerText might be modified by formatters or icons, making it unreliable for database queries.

data-dimension: (New) Stores the column name for the current level (e.g., "category"). This maps the row to the groupOrder array.
